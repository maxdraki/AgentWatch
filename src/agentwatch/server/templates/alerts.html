Quick Setup
import agentwatch

# Alert on health check failures
agentwatch.alerts.on_health_change(handler=my_handler)

# Alert when error rate exceeds 10%
agentwatch.alerts.on_error_rate(threshold_pct=10.0)

# Alert when costs exceed $1/day
agentwatch.alerts.on_cost_threshold(threshold_usd=1.0, period_hours=24)

# Webhook delivery (Slack, Discord, etc.)
agentwatch.alerts.webhook("https://hooks.slack.com/...")

# Check all rules
alerts = agentwatch.alerts.check_all()
